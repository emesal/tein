1//! `(tein time)` — sandbox-safe r7rs time procedures.
2//!
3//! provides:
4//! - `current-second` — wall-clock POSIX time as inexact seconds since epoch
5//! - `current-jiffy` — monotonic nanosecond counter (exact integer)
6//! - `jiffies-per-second` — constant 10⁹
7//! - `timezone-offset-seconds` — local timezone UTC offset in seconds
8//!   (unix: via libc `localtime_r`; windows: via `_get_timezone`; other: always 0)
9
10use std::sync::OnceLock;
11use std::time::{Instant, SystemTime, UNIX_EPOCH};
12use tein_macros::tein_module;
13
14/// monotonic epoch for jiffies — set on first `current-jiffy` call,
15/// constant for the rest of the program run (per r7rs).
16static JIFFY_EPOCH: OnceLock<Instant> = OnceLock::new();
17
18/// query local timezone offset via libc (unix).
19///
20/// uses `localtime_r` to read `tm_gmtoff` from the system timezone database.
21/// returns UTC offset in seconds (e.g. UTC+1 → 3600, UTC-5 → -18000, UTC → 0).
22/// on failure, returns 0 (UTC) as a safe fallback.
23#[cfg(unix)]
24fn local_utc_offset_seconds() -> i64 {
25    use std::ffi::{c_char, c_long};
26    use std::mem::MaybeUninit;
27
28    unsafe extern "C" {
29        fn time(tloc: *mut i64) -> i64;
30        fn localtime_r(timep: *const i64, result: *mut Tm) -> *mut Tm;
31    }
32
33    #[repr(C)]
34    struct Tm {
35        tm_sec: i32,
36        tm_min: i32,
37        tm_hour: i32,
38        tm_mday: i32,
39        tm_mon: i32,
40        tm_year: i32,
41        tm_wday: i32,
42        tm_yday: i32,
43        tm_isdst: i32,
44        tm_gmtoff: c_long,
45        tm_zone: *const c_char,
46    }
47
48    unsafe {
49        let mut t: i64 = 0;
50        time(&mut t);
51        let mut tm = MaybeUninit::<Tm>::uninit();
52        let result = localtime_r(&t, tm.as_mut_ptr());
53        if result.is_null() {
54            return 0;
55        }
56        (*result).tm_gmtoff as i64
57    }
58}
59
60/// query local timezone offset via windows CRT.
61///
62/// `_tzset()` populates the CRT timezone globals from the `TZ` env var or
63/// the system registry.  `_get_timezone()` returns seconds west of UTC
64/// (opposite sign from POSIX `tm_gmtoff`), so we negate.
65/// daylight-saving adjustment is not included (`_get_dstbias` would add it,
66/// but r7rs doesn't require dst-awareness here).
67/// on failure, returns 0 (UTC) as a safe fallback.
68#[cfg(windows)]
69fn local_utc_offset_seconds() -> i64 {
70    unsafe extern "C" {
71        fn _tzset();
72        fn _get_timezone(seconds: *mut i32) -> i32; // errno_t
73    }
74
75    unsafe {
76        _tzset();
77        let mut seconds_west: i32 = 0;
78        if _get_timezone(&mut seconds_west) != 0 {
79            return 0;
80        }
81        -(seconds_west as i64)
82    }
83}
84
85/// timezone offset is unavailable on this platform; returns 0 (UTC).
86#[cfg(not(any(unix, windows)))]
87fn local_utc_offset_seconds() -> i64 {
88    0
89}

91#[tein_module("time")]
92pub(crate) mod time_impl {
93    /// nanoseconds per second — the jiffy resolution constant.
94    #[allow(dead_code)]
95    #[tein_const]
96    pub const JIFFIES_PER_SECOND: i64 = 1_000_000_000;
97
98    /// return current wall-clock time as inexact seconds since the POSIX epoch.
99    ///
100    /// r7rs specifies TAI but explicitly allows UTC with a constant offset.
101    /// we return UTC-based time, matching common implementation practice.
102    #[tein_fn(name = "current-second")]
103    pub fn current_second() -> f64 {
104        super::SystemTime::now()
105            .duration_since(super::UNIX_EPOCH)
106            .expect("system clock before unix epoch")
107            .as_secs_f64()
108    }
109
110    /// return elapsed nanoseconds since a process-relative epoch (exact integer).
111    ///
112    /// the epoch is set on first call and remains constant within a program run,
113    /// per r7rs. uses `Instant` for monotonic timing.
114    #[tein_fn(name = "current-jiffy")]
115    pub fn current_jiffy() -> i64 {
116        let epoch = super::JIFFY_EPOCH.get_or_init(super::Instant::now);
117        epoch.elapsed().as_nanos() as i64
118    }
119
120    /// return local timezone's UTC offset in seconds.
121    ///
122    /// e.g. UTC+1 → 3600, UTC-5 → -18000, UTC → 0.
123    /// platform: unix uses `localtime_r`; windows uses `_get_timezone`;
124    /// other platforms always return 0 (UTC).
125    /// returns real timezone even in sandboxed contexts.
126    #[tein_fn(name = "timezone-offset-seconds")]
127    pub fn timezone_offset_seconds() -> i64 {
128        super::local_utc_offset_seconds()
129    }
