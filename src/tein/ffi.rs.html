1//! raw ffi bindings to chibi-scheme c api
2//!
3//! this module contains unsafe bindings to the underlying chibi-scheme library.
4//! users should prefer the safe wrappers in the parent modules.
5//!
6//! # Safety
7//!
8//! All functions in this module are `unsafe` and require:
9//! - `sexp` pointers must be valid, non-null pointers obtained from chibi-scheme
10//! - `ctx` (context) pointers must be live and not yet destroyed
11//! - String pointers (`*const c_char`) must be valid null-terminated C strings
12//! - Calling functions on invalid or destroyed sexp values is undefined behavior
13//! - The caller must ensure proper memory management across the FFI boundary
14//!
15//! # Intentionally omitted
16//!
17//! `sexp_register_type` / `sexp_register_simple_type` are NOT exposed here.
18//! chibi's C-level type registration ties into the GC finaliser system which
19//! has known bugs (M19-M21 in chibi-scheme-review.md). tein's `ForeignType`
20//! protocol stores objects rust-side, avoiding these issues entirely.
21

536/// extract a rust `String` from a chibi scheme string sexp.
537///
538/// # Safety
539///
540/// caller must ensure `s` is a valid chibi string (`sexp_stringp(s) != 0`).
541#[inline]
542pub unsafe fn sexp_to_rust_string(s: sexp) -> String {
543    unsafe {
544        let ptr = sexp_string_data(s);
545        let len = sexp_string_size(s) as usize;
546        let bytes = std::slice::from_raw_parts(ptr as *const u8, len);
547        String::from_utf8_lossy(bytes).into_owned()
548    }

722/// look up a VFS path, returning the embedded content as a byte slice.
723///
724/// returns `None` if the path is not in the VFS. the returned slice borrows
725/// from static (compiled-in) or thread-local (dynamic) storage and is valid
726/// for the lifetime of the context.
727///
728/// # Safety
729/// The VFS static table and the thread-local dynamic linked list must be
730/// initialised (i.e. called from within a context that has been built).
731#[inline]
732pub unsafe fn vfs_lookup(path: &std::ffi::CStr) -> Option<&[u8]> {
733    unsafe {
734        let mut len: c_uint = 0;
735        let ptr = tein_vfs_lookup(path.as_ptr(), &mut len);
736        if ptr.is_null() {
737            None
738        } else {
739            Some(std::slice::from_raw_parts(ptr as *const u8, len as usize))
740        }
741    }
742}

865/// called from C (`tein_shim.c`) when `tein_vfs_gate == 1`.
866/// checks the module path against the thread-local VFS allowlist and
867/// the filesystem module search path (`FS_MODULE_PATHS`).
868///
869/// two branches:
870///
871/// **VFS branch** — path starts with `/vfs/lib/`:
872/// - `..` traversal guard
873/// - `.scm` passthrough (included file after `.sld` was allowed)
874/// - allowlist prefix matching
875///
876/// **filesystem branch** — any other absolute path:
877/// - `..` traversal guard (fast path before `Path::starts_with`)
878/// - allowed if path is under any dir in `FS_MODULE_PATHS` (canonicalised)
879///
880/// the path arrives as e.g. `/vfs/lib/tein/json.sld`, `/vfs/lib/srfi/69/hash`,
881/// or `/tmp/mylibs/mymod/util.sld`.
882#[unsafe(no_mangle)]
883extern "C" fn tein_vfs_gate_check(path: *const c_char) -> c_int {
884    use crate::sandbox::{FS_MODULE_PATHS, VFS_ALLOWLIST};
885
886    let path_str = unsafe { CStr::from_ptr(path) }.to_str().unwrap_or("");
887
888    // --- VFS path branch ---
889    if let Some(suffix) = path_str.strip_prefix("/vfs/lib/") {
890        // reject path traversal attempts
891        if suffix.contains("..") {
892            return 0;
893        }
894        // .scm passthrough — reachable only after the corresponding .sld was allowed
895        if suffix.ends_with(".scm") {
896            return 1;
897        }
898        // check against the allowlist
899        return VFS_ALLOWLIST.with(|cell| {
900            let list = cell.borrow();
901            if list
902                .iter()
903                .any(|prefix| suffix.starts_with(prefix.as_str()))
904            {
905                1
906            } else {
907                0
908            }
909        });
910    }
911
912    // --- filesystem module path branch ---
913    // reject traversal before Path::starts_with (fast path)
914    if path_str.contains("..") {
915        return 0;
916    }
917    // allow if path is under any configured module search dir.
918    // uses Path::starts_with for proper component-boundary matching
919    // (prevents "/tmp/mylib_evil" matching registered "/tmp/mylib").
920    let path_buf = std::path::Path::new(path_str);
921    FS_MODULE_PATHS.with(|cell| {
922        let dirs = cell.borrow();
923        if dirs.iter().any(|dir| path_buf.starts_with(dir.as_str())) {
924            1
925        } else {
926            0
927        }
928    })
929}

931/// C→rust callback for FS policy enforcement.
932///
933/// Called from `tein_fs_check_access` in `tein_shim.c` when the FS policy
934/// gate is armed (sandboxed contexts). Delegates to `check_fs_access()`
935/// which checks `IS_SANDBOXED` + `FS_POLICY` thread-locals.
936///
937/// Returns 1 (allow) or 0 (deny).
938#[unsafe(no_mangle)]
939extern "C" fn tein_fs_policy_check(path: *const c_char, is_read: c_int) -> c_int {
940    use crate::context::{FsAccess, check_fs_access};
941
942    let path_str = unsafe { CStr::from_ptr(path) }.to_str().unwrap_or("");
943    let access = if is_read != 0 {
944        FsAccess::Read
945    } else {
946        FsAccess::Write
947    };
948    if check_fs_access(path_str, access) {
949        1
950    } else {
951        0
952    }
953}

955/// RAII guard that roots a `sexp` on chibi's global preservatives list.
956///
957/// prevents GC from collecting the guarded object while the guard is alive.
958/// calls `sexp_preserve_object` on creation and `sexp_release_object` on drop,
959/// so early returns and panics are handled automatically.
960///
961/// # safety
962///
963/// `ctx` and `obj` must be valid chibi-scheme pointers.
964/// `obj` must be a heap-allocated sexp (not a fixnum or other immediate value).
965/// the guard must not outlive the context.
966pub struct GcRoot {
967    ctx: sexp,
968    obj: sexp,
969}
970
971impl GcRoot {
972    /// root `obj` in `ctx`'s global preservatives list.
973    ///
974    /// # safety
975    ///
976    /// `ctx` must be a live context. `obj` must be a valid heap-allocated sexp.
977    #[inline]
978    pub unsafe fn new(ctx: sexp, obj: sexp) -> Self {
979        unsafe { sexp_preserve_object(ctx, obj) };
980        Self { ctx, obj }
981    }
982
983    /// the rooted sexp pointer
984    #[inline]
985    pub fn get(&self) -> sexp {
986        self.obj
987    }
988}
989
990impl Drop for GcRoot {
991    fn drop(&mut self) {
992        unsafe { sexp_release_object(self.ctx, self.obj) };
993    }
994}
