1//! Sandboxing and filesystem policy for restricted Scheme environments.
2//!
3//! tein's sandboxing has four independent layers:
4//!
5//! 1. **Module restriction** — importable modules via [`Modules`] + [`ContextBuilder::sandboxed()`](crate::ContextBuilder::sandboxed)
6//! 2. **Step limits** — cap VM instructions per evaluation
7//! 3. **File IO policy** — allowlist filesystem paths for reading/writing
8//! 4. **VFS gate** — restrict `(import ...)` to vetted VFS modules; automatic when using `sandboxed()`
9//!
10//! # Module sets
11//!
12//! [`Modules`] controls which VFS modules sandboxed code can import. The registry
13//! (`VFS_REGISTRY` in `vfs_registry.rs`) declares all vetted modules with their
14//! dependencies, files, and safety tier.
15//!
16//! ```
17//! use tein::{Context, sandbox::Modules};
18//!
19//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
20//! let ctx = Context::builder()
21//!     .standard_env()
22//!     .sandboxed(Modules::Safe)
23//!     .build()?;
24//!
25//! // import and use scheme/base
26//! let result = ctx.evaluate("(import (scheme base)) (+ 1 2)")?;
27//! assert_eq!(result, tein::Value::Integer(3));
28//! # Ok(())
29//! # }
30//! ```
31//!
32//! # File IO policy
33//!
34//! `FsPolicy` controls which filesystem paths Scheme code can access.
35//! Registered internally via
36//! [`ContextBuilder::file_read()`](crate::ContextBuilder::file_read) and
37//! [`ContextBuilder::file_write()`](crate::ContextBuilder::file_write).
38//! Paths are canonicalised before prefix-checking, so symlink and `..`
39//! traversals are resolved.
40//!
41//! Enforcement is at the C opcode level: `eval.c` patches F and G call
42//! `tein_fs_check_access()` before `fopen()` in `sexp_open_input_file_op`
43//! and `sexp_open_output_file_op`. The C dispatcher checks `tein_fs_policy_gate`
44//! (thread-local, 0=off, 1=check) and calls the rust callback
45//! `tein_fs_policy_check` which delegates to [`FsPolicy`] prefix matching.
46//! `file-exists?` and `delete-file` remain rust trampolines (no opcode equivalents).
47//!
48//! # VFS gate
49//!
50//! Module imports in sandboxed contexts are restricted automatically:
51//!
52//! - unsandboxed contexts — no restriction; VFS + filesystem modules all pass.
53//! - sandboxed contexts — only modules in the resolved `Modules` allowlist pass.
54//!   extend with [`.allow_module()`](crate::ContextBuilder::allow_module).
55

61/// Filesystem access policy for sandboxed IO.
62///
63/// Controls which paths Scheme code can read from and write to.
64/// Uses prefix matching against canonicalised paths.
65#[derive(Clone)]
66pub(crate) struct FsPolicy {
67    /// allowed path prefixes for reading
68    pub read_prefixes: Vec<String>,
69    /// allowed path prefixes for writing
70    pub write_prefixes: Vec<String>,
71}
72
73impl FsPolicy {
74    /// Check if a path is allowed for reading.
75    ///
76    /// Canonicalises the full path (file must exist for reads).
77    /// Returns false if path is invalid or canonicalisation fails.
78    pub fn check_read(&self, path: &str) -> bool {
79        Path::new(path)
80            .canonicalize()
81            .ok()
82            .map(|canon| {
83                let canon_str = canon.to_string_lossy();
84                self.read_prefixes
85                    .iter()
86                    .any(|prefix| canon_str.starts_with(prefix))
87            })
88            .unwrap_or(false)
89    }
90
91    /// Check if a path is allowed for writing.
92    ///
93    /// Canonicalises the parent directory (must exist), appends filename.
94    /// The file itself doesn't need to exist (R7RS: open-output-file creates it).
95    pub fn check_write(&self, path: &str) -> bool {
96        let p = Path::new(path);
97        let parent = match p.parent().and_then(|d| d.canonicalize().ok()) {
98            Some(d) => d,
99            None => return false,
100        };
101        let filename = match p.file_name() {
102            Some(f) => f,
103            None => return false,
104        };
105        let full = parent.join(filename);
106        let full_str = full.to_string_lossy();
107        self.write_prefixes
108            .iter()
109            .any(|prefix| full_str.starts_with(prefix))
110    }
111}

177/// resolve transitive deps from `VFS_REGISTRY`.
178///
179/// follows `deps` recursively for each entry, returns a deduplicated flat list
180/// of all module path strings (including the inputs). unknown paths are included
181/// as-is (not expanded).
182pub fn registry_resolve_deps(paths: &[&str]) -> Vec<String> {
183    let mut result: Vec<String> = Vec::new();
184    let mut seen: std::collections::HashSet<&str> = std::collections::HashSet::new();
185    let mut stack: Vec<&str> = paths.to_vec();
186
187    while let Some(path) = stack.pop() {
188        if !seen.insert(path) {
189            continue;
190        }
191        result.push(path.to_string());
192
193        // union deps from all entries with this path (handles Embedded + Shadow pairs)
194        for entry in VFS_REGISTRY.iter().filter(|e| e.path == path) {
195            for dep in entry.deps {
196                if !seen.contains(dep) {
197                    stack.push(dep);
198                }
199            }
200        }
201    }
202
203    result
204}

206/// build the default safe allowlist from `VFS_REGISTRY` (`default_safe: true` entries
207/// with their transitive deps resolved, filtered by active cargo features).
208pub(crate) fn registry_safe_allowlist() -> Vec<String> {
209    let paths: Vec<&str> = VFS_REGISTRY
210        .iter()
211        .filter(|e| e.default_safe && feature_enabled(e.feature))
212        .map(|e| e.path)
213        .collect();
214    registry_resolve_deps(&paths)
215}

247/// Inject VFS shadow modules for sandboxed contexts.
248///
249/// Iterates `VFS_REGISTRY` for `VfsSource::Shadow` entries and registers
250/// their `.sld` content into the dynamic VFS under canonical `/vfs/lib/`
251/// paths. Hand-written shadows use `shadow_sld`; generated stubs (from
252/// `SHADOW_STUBS` via build.rs) are looked up in `GENERATED_SHADOW_SLDS`.
253///
254/// Must be called before the VFS gate is armed (before `VFS_GATE` is set
255/// to `GATE_CHECK`).
256pub(crate) fn register_vfs_shadows() -> crate::error::Result<()> {
257    use std::ffi::CString;
258
259    let register_one = |path: &str, sld: &str| -> crate::error::Result<()> {
260        let vfs_path = format!("/vfs/lib/{}.sld", path);
261        let c_path = CString::new(vfs_path)
262            .map_err(|_| crate::error::Error::InitError("VFS path contains null bytes".into()))?;
263        let rc = unsafe {
264            crate::ffi::tein_vfs_register(
265                c_path.as_ptr(),
266                sld.as_ptr() as *const std::ffi::c_char,
267                sld.len() as std::ffi::c_uint,
268            )
269        };
270        if rc != 0 {
271            return Err(crate::error::Error::InitError(
272                "VFS shadow registration failed: out of memory".into(),
273            ));
274        }
275        Ok(())
276    };
277
278    for entry in VFS_REGISTRY.iter() {
279        if entry.source != VfsSource::Shadow {
280            continue;
281        }
282        if !feature_enabled(entry.feature) {
283            continue;
284        }
285        if let Some(sld) = entry.shadow_sld {
286            // hand-written shadow (scheme/file, scheme/process-context, etc.)
287            register_one(entry.path, sld)?;
288        }
289        // generated stubs have shadow_sld: None — handled below
290    }
291
292    // generated stubs from SHADOW_STUBS (via build.rs)
293    for &(path, sld) in GENERATED_SHADOW_SLDS.iter() {
294        register_one(path, sld)?;
295    }
296
297    Ok(())
298}

306/// look up the exported binding names for a VFS module by path.
307///
308/// returns `Some(&[...])` if the module is in the generated exports table,
309/// or `None` for paths not in the registry (e.g. `chibi/*` internal modules).
310#[cfg_attr(not(test), allow(dead_code))]
311pub(crate) fn module_exports(path: &str) -> Option<&'static [&'static str]> {
312    MODULE_EXPORTS
313        .iter()
314        .find(|(p, _)| *p == path)
315        .map(|(_, exports)| *exports)
316}

318/// collect all exports from modules NOT in the given allowlist.
319///
320/// returns `(binding_name, module_path)` pairs for registering UX stubs —
321/// informative errors that tell sandbox users which module to import.
322/// bindings from modules with empty export lists (alias modules like `scheme/bitwise`)
323/// are silently skipped since they have no top-level names to stub.
324pub(crate) fn unexported_stubs(allowed_modules: &[String]) -> Vec<(&'static str, &'static str)> {
325    // collect every binding name already provided by an allowed module.
326    // stubs must never be generated for these — doing so clobbers real bindings.
327    // this is especially important for mega re-export bundles like scheme/red and
328    // scheme/small that duplicate hundreds of names from other modules.
329    let covered: std::collections::HashSet<&str> = MODULE_EXPORTS
330        .iter()
331        .filter(|(p, _)| allowed_modules.iter().any(|a| a == p))
332        .flat_map(|(_, exports)| exports.iter().copied())
333        .collect();
334
335    let mut stubs = Vec::new();
336    for (path, exports) in MODULE_EXPORTS.iter() {
337        if !allowed_modules.iter().any(|a| a == path) {
338            for name in exports.iter() {
339                if !covered.contains(name) {
340                    stubs.push((*name, *path));
341                }
342            }
343        }
344    }
345    stubs
346}

348/// Module set configuration for sandboxed contexts.
349///
350/// Controls which VFS modules are importable when using [`ContextBuilder::sandboxed()`].
351/// Dependencies are always resolved automatically from the registry.
352///
353/// # examples
354///
355/// ```
356/// use tein::{Context, sandbox::Modules};
357///
358/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
359/// // allow only scheme/base
360/// let ctx = Context::builder()
361///     .standard_env()
362///     .sandboxed(Modules::only(&["scheme/base"]))
363///     .build()?;
364/// let result = ctx.evaluate("(import (scheme base)) (+ 1 2)")?;
365/// assert_eq!(result, tein::Value::Integer(3));
366/// # Ok(())
367/// # }
368/// ```
369#[derive(Clone, Debug, Default)]
370pub enum Modules {
371    /// conservative safe set — default for sandboxed contexts.
372    ///
373    /// includes all modules marked `default_safe: true` in the registry,
374    /// with transitive deps resolved. excludes `scheme/eval` and `scheme/r5rs`.
375    /// `scheme/time` is available via a shadow re-exporting `(tein time)` (requires
376    /// feature `"time"`). `scheme/repl`, `scheme/file`, `scheme/load`,
377    /// `scheme/process-context`, and `tein/process` are included via
378    /// shadow modules or neutered trampolines.
379    #[default]
380    Safe,
381    /// all vetted modules in the registry (superset of `Safe`).
382    All,
383    /// syntax only — no modules, not even `scheme/base`.
384    ///
385    /// `import` is still available as syntax (so code can attempt imports),
386    /// but all module imports will be rejected by the VFS gate.
387    None,
388    /// custom explicit module list; transitive deps resolved automatically.
389    Only(Vec<String>),
390}
391
392impl Modules {
393    /// construct a custom module list from module path strings.
394    ///
395    /// transitive dependencies are resolved automatically at build time.
396    pub fn only(modules: &[&str]) -> Self {
397        Modules::Only(modules.iter().map(|s| s.to_string()).collect())
398    }
399}

401/// check whether a cargo feature gate is satisfied at compile time.
402///
403/// **keep in sync with `feature_enabled` in `build.rs`** — both must be updated
404/// when adding or removing cargo features. they can't be merged because `cfg!` resolves
405/// differently in build script vs lib contexts.
406#[inline]
407fn feature_enabled(feature: Option<&str>) -> bool {
408    match feature {
409        None => true,
410        Some("json") => cfg!(feature = "json"),
411        Some("toml") => cfg!(feature = "toml"),
412        Some("uuid") => cfg!(feature = "uuid"),
413        Some("time") => cfg!(feature = "time"),
414        Some("regex") => cfg!(feature = "regex"),
415        Some("crypto") => cfg!(feature = "crypto"),
416        Some("http") => cfg!(feature = "http"),
417        Some(f) => {
418            // unknown feature name — conservatively include (build.rs handles gating)
419            eprintln!("warning: unknown feature gate in VFS_REGISTRY: {f}");
420            true
421        }
422    }
423}

634            !stubs.iter().any(|(name, _)| *name == "+"),
635            "'+' covered by allowed scheme/base must not appear in stubs from any module \
636             (e.g. scheme/red)"
637        );
638        // a binding unique to scheme/red (not in scheme/base) should still produce a stub
639        let red_unique = stubs
640            .iter()
641            .any(|(_, module)| *module == "scheme/red" || *module == "scheme/small");
642        assert!(
643            red_unique,
644            "scheme/red or scheme/small should still contribute stubs for names \
645             not covered by the allowlist"
646        );
