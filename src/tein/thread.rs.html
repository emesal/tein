1//! shared protocol types for thread-based context wrappers
2//!
3//! both [`TimeoutContext`](crate::TimeoutContext) and
4//! [`ThreadLocalContext`](crate::managed::ThreadLocalContext) run a
5//! [`Context`](crate::Context) on a dedicated thread and communicate
6//! via channels. this module contains the shared request/response
7//! protocol and the `SendableValue` wrapper.
8
9use crate::Value;
10use crate::error::Result;
11
12/// function pointer type for variadic foreign functions
13pub(crate) type ForeignFnPtr = unsafe extern "C" fn(
14    crate::ffi::sexp,
15    crate::ffi::sexp,
16    crate::ffi::sexp_sint_t,
17    crate::ffi::sexp,
18) -> crate::ffi::sexp;
19
20/// request sent to a context thread
21pub(crate) enum Request {
22    /// evaluate a string of scheme code
23    Evaluate(String),
24    /// call a procedure with arguments
25    Call(SendableValue, Vec<SendableValue>),
26    /// register a variadic foreign function
27    DefineFnVariadic {
28        /// scheme name for the function
29        name: String,
30        /// the raw function pointer
31        f: ForeignFnPtr,
32    },
33    /// rebuild the context from the stored builder + init closure
34    Reset,
35    /// shut down the context thread
36    Shutdown,
37}
38
39// SAFETY: Request contains SendableValue which wraps Value (may hold raw
40// sexp pointers). safe because values only travel to the context thread
41// where the context that created them lives.
42unsafe impl Send for Request {}
43
44/// response from a context thread
45pub(crate) enum Response {
46    /// result of evaluate or call
47    Value(Result<Value>),
48    /// result of define_fn_variadic
49    Defined(Result<()>),
50    /// result of reset
51    Reset(Result<()>),
52}
53
54// SAFETY: Response contains Result<Value> which may hold Value::Procedure
55// (a raw *mut c_void). safe because values only travel between the caller
56// and the single context thread — Procedure pointers are only
57// dereferenced on the context thread where the context lives.
58unsafe impl Send for Response {}
59
60/// wrapper allowing a Value to be sent across threads
61///
62/// # safety
63/// safe because values are only ever sent *back* to the thread that owns
64/// the context. Procedure values contain raw sexp pointers that are only
65/// valid on the context thread — this wrapper ensures they travel back
66/// to where they came from.
67pub(crate) struct SendableValue(pub(crate) Value);
68
69// SAFETY: see struct-level doc. values only travel between the caller
70// and the single context thread, and Procedure pointers are only
71// dereferenced on the context thread.
72unsafe impl Send for SendableValue {}
